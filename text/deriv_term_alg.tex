\section{Terminal algebra $W_N$}


Denote $\beta^i_{jk} := \alpha^i_{jk} + \alpha^i_{kj}$.

\vspace{2mm}

Then $\{ \beta^i_{jk} ~|~ j \leq k\}$ is a basis of $W_N$, and $\text{dim} W_N = N^2 + \frac{N^3 - N^2}{2} = \frac{1}{2}N^2 (N+1)$.

The multiplication table of $W_N$ is the following

\vspace{2mm}

$
\beta^i_{jk} \cdot \beta^l_{mn} =
\delta_{j1} ( \delta_{kl} \beta^i_{mn} -  \delta_{mi} \beta^l_{kn} -  \beta_{ni} \beta^l_{mk}) 
+ \delta_{k1} ( \delta_{jl} \beta^i_{mn} -  \delta_{mi} \beta^l_{jn} -  \beta_{ni} \beta^l_{mj}).
$

\vspace{3mm}

If
$\be_{ijk} \cdot \be_{lmn} = \sum\limits_{pqr} \lambda^{pqr}_{ijklmn} \be_{pqr}$,

then

$\lambda^{pqr}_{ijklmn} = \tilde{\lambda}^{pqr}_{ijklmn}$ if $q=r$,

and $\lambda^{pqr}_{ijklmn} = \tilde{\lambda}^{pqr}_{ijklmn} + \tilde{\lambda}^{prq}_{ijklmn}
= 2 \tilde{\lambda}^{pqr}_{ijklmn} $ if $q \neq r$,

where

$\tilde{\lambda}^{pqr}_{ijklmn} 
= \d_{j1} (\d_{kl} \d_{pi} \d_{qm} \d_{rn}
- \d_{mi} \delta_{pl} \d_{qk} \d_{rn}
- \d_{ni} \d_{pl} \d_{qm} \d_{rk})
+ \d_{k1} (\d_{jl} \d_{pi} \d_{qm} \d_{rn}
- \d_{mi} \d_{pl} \d_{qj} \d_{rn}
- \d_{ni} \d_{pl} \d_{qm} \d_{rj}).
$

\vspace{3mm}

i.e. $\lambda^{pqr}_{ijklmn} = (2 - \d_{qr}) \tilde{\lambda}^{pqr}_{ijklmn}$.


\subsection{Algebra of derivations}

Denote $\beta_L := \beta_{ijk} = \beta^i_{jk}$, 

where correspondence between $L$ and $ijk$:

$L=1$ --- $(ijk) = (1,1,1)$,

$L=2$ --- $(ijk) = (1,1,2)$,

$L=3$ --- $(ijk) = (1,2,2)$,

$L=4$ --- $(ijk) = (2,1,1)$,

$L=5$ --- $(ijk) = (2,1,2)$,

$L=6$ --- $(ijk) = (2,2,2)$,

\vspace{5mm}

Let $D(\be_L) = \sum\limits_S d^S_L \be_S$, and  $\be_I \cdot \be_L = \sum\limits_P \lambda^P_{IL} \be_P$, where $\lambda^P_{IL} := \lambda^{pqr}_{ijklmn}$.

\vspace{2mm}

It follows from $D(\be_I) \cdot \be_L + \be_I \cdot D(\be_L) - D(\be_I \cdot \be_L) = 0$ 

\vspace{2mm}

that $\sum\limits_S ( \lambda^P_{SL} d^S_I + \lambda^P_{IS} d^S_L - \lambda^S_{IL} d^P_S) = 0$ for any $P, I, L$.
